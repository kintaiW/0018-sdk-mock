代码示例

#include "sdf.h"

int main() {
    void *hDevice = NULL, *hSession = NULL;

    /* 1. 打开设备与会话 */
    SDF_OpenDevice(&hDevice);
    SDF_OpenSession(hDevice, &hSession);

    /* 2. SM3 哈希 */
    SDF_HashInit(hSession, SGD_SM3, NULL, NULL, 0);
    SDF_HashUpdate(hSession, (const unsigned char*)"hello", 5);
    unsigned char hash[32]; unsigned int hlen;
    SDF_HashFinal(hSession, hash, &hlen);

    /* 3. SM2 临时密钥对签名 */
    ECCrefPublicKey pub; ECCrefPrivateKey pri;
    SDF_GenerateKeyPair_ECC(hSession, SGD_SM2_1, 256, &pub, &pri);
    ECCSignature sig;
    SDF_ExternalSign_ECC(hSession, SGD_SM2_1, &pri, hash, 32, &sig);
    SDF_ExternalVerify_ECC(hSession, SGD_SM2_1, &pub, hash, 32, &sig);

    /* 4. 清理 */
    SDF_CloseSession(hSession);
    SDF_CloseDevice(hDevice);
    return 0;
}
use sdf_mock::sdf_impl::{device::*, asymmetric::*, hash::*};
use sdf_mock::types::{alg_id, ECCrefPublicKey, ECCrefPrivateKey, ECCSignature};

fn main() {
    // 1. 打开设备与会话
    assert_eq!(sdf_open_device(), 0);
    let mut session = 0u32;
    assert_eq!(sdf_open_session(&mut session), 0);

    // 2. SM3 哈希
    sdf_hash_init(session, alg_id::SGD_SM3, None, b"");
    sdf_hash_update(session, b"hello");
    let mut hash = [0u8; 32];
    sdf_hash_final(session, &mut hash);

    // 3. SM2 签名
    let mut pub_key = ECCrefPublicKey::default();
    let mut pri_key = ECCrefPrivateKey::default();
    sdf_generate_key_pair_ecc(session, alg_id::SGD_SM2_1, 256,
                               &mut pub_key, &mut pri_key);
    let mut sig = ECCSignature::default();
    sdf_external_sign_ecc(session, alg_id::SGD_SM2_1, &pri_key, &hash, &mut sig);

    // 4. 清理
    sdf_close_session(session);
    sdf_close_device();
}
